import tkinter as tk import keyboard class OverlayText: def __init__(self, text="Я из Coalition of Colonies с планеты Himeo (United Syndicates of Himeo) из города Dorshafen\n" "Таяры - Tajara\n" "Скреллы - Skrells\n" "Унати - Unathi\n" "Дионы - Dionae\n" "Варуки - Vaurca\n" "ИПС - IPC\n" "Роботы - Synths"): self.root = tk.Tk() self.root.attributes('-topmost', True) self.root.attributes('-alpha', 0.8) self.root.overrideredirect(True) bg_color = '#222222' self.root.configure(bg=bg_color) self.text_widget = tk.Text( self.root, bg=bg_color, fg='white', font=('Futura Md BT [Rus by me]', 24, 'bold'), bd=0, highlightthickness=0, wrap='word' ) self.text_widget.insert('1.0', text) self.text_widget.config(state='disabled') self.text_widget.pack(padx=10, pady=10) self.root.update_idletasks() screen_width = self.root.winfo_screenwidth() screen_height = self.root.winfo_screenheight() self.root.geometry(f"{screen_width}x{screen_height}+0+0") self.visible = True self.root.withdraw() keyboard.add_hotkey('right shift', self.toggle_overlay) self.root.after(100, self.check_hotkey) self.root.mainloop() def toggle_overlay(self): if self.visible: self.root.withdraw() else: self.root.deiconify() self.visible = not self.visible def check_hotkey(self): self.root.after(100, self.check_hotkey) if __name__ == "__main__": OverlayText()
